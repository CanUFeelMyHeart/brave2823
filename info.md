git - программа контроля версий над проектом

Возможность хранить различные версии проекта
Возможность возвращаться к различным версиям проекта

Сохраняет в памяти не файлы целиком, а разницу между файлами

cd .. - выйти к папке выше в локалке

cls // git clear // очистка терминала

git --version //проверка настройки гита

**git config --global [user.name](http://user.name) //**проверка настройки гита

**git config --global user.email //**проверка настройки гита

**git status //**получить информацию от git о его текущем состоянии, проверка наличия репозитория

**git init //** создание локального репозитория

git add . // добавить файл или файлы к следующему коммиту

git commit -m “message” //создание коммита.

git commit -a -m “текст_коммита”//  создание изменения и его коммит (используется только с сохраненными файлами: “**git add .** ”

git commit —amend -m //"Иcправленный заголовок”

git branch //посмотреть список веток в репозитория
git branch <название ветки>// создать новую ветку

git checkout <название ветки> / либо 4 знака коммита// переход к другой ветке

git checkout -b branch_name  //  создание ветки и переход  в нее

git branch -d <название ветки>//удалить ветку, d - проверяет есть ли изменения для merge (слияния)

git branch -D <название ветки>// удалить ветку невзирая на merge

git checkout master//вернуться к актуальному состоянию и продолжить работу

git merge view_branch // Чтобы слить любую ветку (указываем имя) с текущей,слияние веток -необходимо коммитить

git log - журнал коммитов // зажать enter - до конца - выход из перечня -\*\* Q

git log --graph// журнал коммитов с визуализацией **- зажать enter - до конца - выход из перечня -** Q

git log --oneline // сокращенный вывод всех коммитов\*\*

git log --oneline - - graph // сокращенный вывод всех коммитов + схемы / график\*\*

git diff //показывает разницу м/д сохраненным файлом и рабочим

git ignore //исключить ненужные файлы из загрузки

git reset --soft HEAD~1 // удалить 1 коммит, но оставить файлы

git reset --hard HEAD~2 // удалить 2 последних коммита и все файлы

git switch -c <branch-name> //переключиться на созданную новую ветку

git switch <branch-name> // создать ветку

git cherry-pick // Примените изменения, внесенные некоторыми существующими коммитами?

git revert //это операция для безопасной отмены изменений, действие которой направлено в будущее.
Для отката изменений команда не удаляет из истории коммиты или родительские элементы,
a создает новый коммит с отменой нужных действий.

git rebase //повторно применяет коммиты поверх другой базовой ветки? (отличия от merge - слияние сохраняет историю в первозданном виде, а перебазирование ее перезаписывает)

git cd //указать путь папки с репозиторием - cd пробел tab / нажимаем на папку /открыть во встроенном терминале.

git pull // Эта команда позволяет скачать все из текущего репозитория и автоматически
сделать merge с нашей версией

git push // эта команда позволяет отправить нашу версию репозитория на внешний
